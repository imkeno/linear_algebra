def matxMultiply(A, B):
    row_a,column_a = shape(A)
    row_b,column_b = shape(B)
    #print("row_a = {}, column_a = {},row_b = {}, column_b = {}".format(row_a,column_a,row_b,column_b))
    AB = [[0]*column_b for i in range(row_a)]
    if column_a != row_b:
        raise ValueError("ValueError")
    else:
        for i in range(row_a):
            for j in range(column_b):
                for k in range(column_a):
                    AB[i][j] += A[i][k]*B[k][j]
        return AB

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    """
    1 Ab
    2 Ab
    3 Ab 
        Ab
        c,c  c~N
        if 0
            ANone
        else
            c 
            c1
            c0
            
    4 Ab
    """
    #Ab
    if(len(A) != len(b)):
        return None
    #Ab
    Ab = augmentMatrix(A,b)
    #Ab
    length = len(A)
    for c in range(length):
        #print(c)
        trans_Ab = transpose(Ab)
        cur_row = trans_Ab[c][c:]
        max_value = max(cur_row,key=abs)
        #0
        if abs(max_value) < epsilon:
            return None #A
        else:
            max_row = cur_row.index(max_value) + c
            swapRows(Ab,c,max_row)
            scaleRow(Ab,c,1./Ab[c][c])#c1
            for i in range(length): #c0
                if(i != c):
                    addScaledRow(Ab,i,c,-Ab[i][c]) #c0
    matxRound(Ab,decPts)
    x = [[i[-1]] for i in Ab]
    return x

3.3 (选做) 找到参数 $m, b$ 使得平方平均误差最小¶

( $ (x^2)' = 2x $ )

$m, b$

$E$ $$ E = \frac{1}{2}\sum_{i=1}^{n}{(y_i - mx_i - b)^2} $$

$E = \frac{n}{2}MSE$, $E$ $MSE$ $E$ $m, b$ $E$ $m$, $E$ $b$ 0.

$$ \begin{cases} \displaystyle \frac{\partial E}{\partial m} =0 \\ \\ \displaystyle \frac{\partial E}{\partial b} =0 \\ \end{cases} $$

3.3.1 计算目标函数相对于参数的导数¶

/ $$ \frac{\partial E}{\partial m} = \sum_{i=1}^{n}{-x_i(y_i - mx_i - b)} $$

$$ \frac{\partial E}{\partial b} = \sum_{i=1}^{n}{-(y_i - mx_i - b)} $$

:

$$ \frac{\partial E}{\partial m} = \frac{1}{2}\sum_{i=1}^{n}(\frac{\partial (y_i^-b)^2)}{\partial(m) } + \frac{\partial (m^2x_i^2)}{\partial(m) }-\frac{\partial (2(y_i-b)x_im)}{\partial(m) }) $$

$$ \frac{\partial E}{\partial m} = \frac{1}{2}\sum_{i=1}^{n}(2mx_i^2 - 2(y_i-b)x_i) $$

$$ \frac{\partial E}{\partial m} = \sum_{i=1}^{n}(x_i(mx_i - y_i+b) $$

$$ \frac{\partial E}{\partial m} = \sum_{i=1}^{n}(-x_i( y_i-mx_i-b) $$

$$ \frac{\partial E}{\partial b} = \frac{1}{2}\sum_{i=1}^{n}(\frac{\partial (y_i^-mx_i)^2)}{\partial(b) } + \frac{\partial (b^2)}{\partial(b) }-\frac{\partial (2b(y_i- mx_i)}{\partial(b) }) $$

$$ \frac{\partial E}{\partial b} = \frac{1}{2}\sum_{i=1}^{n}(2b -2(y_i-mx_i)) $$

$$ \frac{\partial E}{\partial b} = \sum_{i=1}^{n}-(y_i-mx_i-b) $$

3.3.2 实例推演¶

$$ \begin{cases} \displaystyle \sum_{i=1}^{n}{-x_i(y_i - mx_i - b)} =0 \\ \\ \displaystyle \sum_{i=1}^{n}{-(y_i - mx_i - b)} =0 \\ \end{cases} $$
